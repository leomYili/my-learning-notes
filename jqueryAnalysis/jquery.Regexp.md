## jquery中的正则
```
// 验证是否是html字符串或者带#的id选择器
rquickExpr = /^(?:\s*(<[\w\W]+>)[^>]*|#([\w-]+))$/

// 验证是否是独立的标签,或者是完整的标签
rsingleTag = (/^<([a-z][^\/\0>:\x20\t\r\n\f]*)[\x20\t\r\n\f]*\/?>(?:<\/\1>|)$/i);

// 匹配\x20,\t,\r,\n,\f分割的字符
rnothtmlwhite = (/[^\x20\t\r\n\f]+/g)

// 类数组判断
var rbrace = /^(?:\{[\w\W]*\}|\[[\w\W]*\])$/
```
